% --- Glosario (términos) ---
\newglossaryentry{standalone}{
	name={stand alone},
	description={Propiedad de un sistema que puede ejecutarse de forma autónoma en un entorno local, sin depender de servicios externos o conexión a internet}
}

\newglossaryentry{recall}{
	name={Recall (sensibilidad)},
	description={Métrica que cuantifica la proporción de casos reales de una clase identificados correctamente por el modelo; se calcula como TP/(TP+FN)}
}

\newglossaryentry{truncatedsvd}{
	name={TruncatedSVD},
	description={Técnica de reducción de dimensionalidad que aproxima la descomposición en valores singulares conservando solo k componentes principales; se aplica frecuentemente sobre matrices TF-IDF para obtener representaciones latentes}
}

\newglossaryentry{svm}{
	name={svm},
	description={Máquina de Vectores de Soporte, algoritmo de aprendizaje automático supervisado utilizado para tareas de clasificación. Su objetivo es separar las clases mediante una frontera de decisión lo más clara posible}
}
\newglossaryentry{regresion_logistica}{
	name={Regresión logística},
	description={Modelo de clasificación supervisada que estima la probabilidad de pertenencia a una clase a partir de una combinación lineal de características}
}

\newglossaryentry{naive_bayes}{
	name={Naive Bayes},
	description={Familia de clasificadores probabilísticos basada en el teorema de Bayes y en la asunción de independencia condicional entre características}
}

% --- Glosario ConceptTeorico
\newglossaryentry{mlp}{
	name=MLP,
	description={Perceptrón Multicapa (\textit{Multi-layer Perceptron}). Red neuronal artificial feedforward que captura relaciones no lineales mediante capas ocultas y entrenamiento por retropropagación}
}

\newglossaryentry{gscv}{
	name=GridSearchCV,
	description={Búsqueda de rejilla con validación cruzada (\textit{Grid Search Cross-Validation}). Procedimiento de exploración estructurada sobre un conjunto discreto de hiperparámetros para encontrar la combinación óptima}
}

\newglossaryentry{stemming}{
	name=Stemming,
	description={Proceso lingüístico de reducción de una palabra a su raíz o tallo (\textit{stem}). Técnica de bajo coste computacional comparada con la lematización}
}

\newglossaryentry{lsa}{
	name=LSA,
	description={Análisis Semántico Latente (\textit{Latent Semantic Analysis}). Técnica de procesamiento de lenguaje natural que utiliza SVD para identificar patrones en las relaciones entre términos y conceptos}
}

% El comando \figura nos permite insertar figuras comodamente, y utilizando
% siempre el mismo formato. Los parametros son:
% 1 --> Porcentaje del ancho de página que ocupará la figura (de 0 a 1)
% 2 --> Fichero de la imagen
% 3 --> Texto a pie de imagen
% 4 --> Etiqueta (label) para referencias
% 5 --> Opciones que queramos pasarle al \includegraphics
% 6 --> Opciones de posicionamiento a pasarle a \begin{figure}
	\newcommand{\figuraConPosicion}[6]{%
		\setlength{\anchoFloat}{#1\textwidth}%
		\addtolength{\anchoFloat}{-4\fboxsep}%
		\setlength{\anchoFigura}{\anchoFloat}%
		\begin{figure}[#6]
			\begin{center}%
				\Ovalbox{%
					\begin{minipage}{\anchoFloat}%
						\begin{center}%
							\includegraphics[width=\anchoFigura,#5]{#2}%
							\caption{#3}%
							\label{#4}%
						\end{center}%
					\end{minipage}
				}%
			\end{center}%
		\end{figure}%
	}
